#include "IRremote.h"
#include <IRremoteInt.h>
#include <ir_Lego_PF_BitStreamEncoder.h>

const int buttonPin = 4;
int buttonState = 0;
IRsend irsend;

void setup() {
  pinMode(buttonPin,INPUT);
}

void loop() {
  buttonState = digitalRead(buttonPin);
  if(buttonState == HIGH){
    irsend.sendNEC(0x4FB48B7,32);
  }
}
